#把trx解包工具里的modify文件夹放进/root/

#把padavan.trx放进/root/modify/

#解包
./modify.sh e padavan.trx

#把/etc_ro/v2ray文件夹复制进/root/modify/squashfs-root/etc_ro/

#重新封包为newrom.trx
./modify.sh c newrom.trx
